#可传入任何参数
def function(*args,**kwargs):
    pass

# 不定长参数，可传入多个参数，传入的值组成一个元组tuple
def function2(*args):
    pass

# 传入的参数为多个关键字参数,组成一个字典dict
def function3(**kwargs):
    pass
function3(key1=a,key2=b,key3=c)

# 匿名函数Lamda
# lamda只能有一行
# lamda arg1,arg2,……argn:函数表达式
sum=lamda num1, num2:num1+num2
print(sum(1,2))

def func(*arg, **kw):
    for i in arg:
        for k, v in kw.items():
            if i == k:
                print('My name is {k},I am {v} years old.'.format(k=k, v=v))

str = ['tom','eddy']
kw = {'eddy': 18, 'tom': 20}

func(*str, **kw)


# 输出结果：
# My name is tom,I am 20 years old.
# My name is eddy,I am 18 years old.
